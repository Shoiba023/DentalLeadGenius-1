const resend = new Resend(process.env.RESEND_API_KEY);

export async function sendEmail({ to, subject, html }) {

      from: "DentalLeadGenius <support@dentalleadgenius.com>",

      subject,
      html,

    return { ok: false, error: err.message };
  }
}
